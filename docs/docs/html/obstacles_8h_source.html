298  // implements checkLineIntersection() of the base class
299  bool checkLineIntersection(const Eigen::Vector2d& line_start, const Eigen::Vector2d& line_end, double min_dist = 0) const override {
300  // Distance Line - Circle
301  // refer to http://www.spieleprogrammierer.de/wiki/2D-Kollisionserkennung#Kollision_Kreis-Strecke
302  Eigen::Vector2d a = line_end - line_start; // not normalized! a=y-x
303  Eigen::Vector2d b = pos_ - line_start; // b=m-x
304 
305  // Now find nearest point to circle v=x+a*t with t=a*b/(a*a) and bound to 0<=t<=1
306  double t = a.dot(b) / a.dot(a);
307  if (t < 0) {
308  t = 0;
309  } // bound t (since a is not normalized, t can be scaled between 0 and 1 to parametrize the line
310  else if (t > 1) {
311  t = 1;
312  }
313  Eigen::Vector2d nearest_point = line_start + a * t;
314 
315  // check collision
316  return checkCollision(nearest_point, min_dist);
317  }

648  // implements checkCollision() of the base class
649  bool checkCollision(const Eigen::Vector2d& point, double min_dist) const override {
650  // line case
651  if (noVertices() == 2) {
652  return getMinimumDistance(point) <= min_dist;
653  }
654 
655  // check if point is in the interior of the polygon
656  // point in polygon test - raycasting (http://www.ecse.rpi.edu/Homepages/wrf/Research/Short_Notes/pnpoly.html)
657  // using the following algorithm we may obtain false negatives on edge-cases, but that's ok for our purposes
658  int i;
659  int j;
660  bool c = false;
661  for (i = 0, j = noVertices() - 1; i < noVertices(); j = i++) {
662  if (((vertices_.at(i).y() > point.y()) != (vertices_.at(j).y() > point.y())) &&
663  (point.x() < (vertices_.at(j).x() - vertices_.at(i).x()) * (point.y() - vertices_.at(i).y()) / (vertices_.at(j).y() - vertices_.at(i).y()) + vertices_.at(i).x())) {
664  {
665  c = !c;
666  }
667  }
668  }
669  if (c > 0) {
670  return true;
671  }
672 
673  // If this statement is reached, the point lies outside the polygon or maybe on its edges
674  // Let us check the minium distance as well
675  return min_dist == 0 ? false : getMinimumDistance(point) < min_dist;
676  }
