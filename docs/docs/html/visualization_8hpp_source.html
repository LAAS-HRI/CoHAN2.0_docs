46 template <typename GraphType>
47 void TebVisualization::publishGraph(const GraphType& graph, const std::string& ns_prefix) {
48  if (printErrorWhenNotInitialized()) return;
49 
50  using GraphVertexIterator = typename boost::graph_traits<GraphType>::vertex_iterator;
51  using GraphEdgeIterator = typename boost::graph_traits<GraphType>::edge_iterator;
52 
53  // Visualize Edges
54  visualization_msgs::Marker marker;
55  marker.header.frame_id = cfg_->map_frame;
56  marker.header.stamp = ros::Time::now();
57  marker.ns = ns_prefix + "Edges";
58  marker.id = 0;
59 // #define TRIANGLE
60 #ifdef TRIANGLE
61  marker.type = visualization_msgs::Marker::TRIANGLE_LIST;
62 #else
63  marker.type = visualization_msgs::Marker::LINE_LIST;
64 #endif
65  marker.action = visualization_msgs::Marker::ADD;
66 
67  GraphEdgeIterator it_edge;
68  GraphEdgeIterator end_edges;
69  for (boost::tie(it_edge, end_edges) = boost::edges(graph); it_edge != end_edges; ++it_edge) {
70 #ifdef TRIANGLE
71  geometry_msgs::Point point_start1;
72  point_start1.x = graph[boost::source(*it_edge, graph)].pos[0] + 0.05;
73  point_start1.y = graph[boost::source(*it_edge, graph)].pos[1] - 0.05;
74  point_start1.z = 0;
75  marker.points.push_back(point_start1);
76  geometry_msgs::Point point_start2;
77  point_start2.x = graph[boost::source(*it_edge, graph)].pos[0] - 0.05;
78  point_start2.y = graph[boost::source(*it_edge, graph)].pos[1] + 0.05;
79  point_start2.z = 0;
80  marker.points.push_back(point_start2);
81 
82 #else
83  geometry_msgs::Point point_start;
84  point_start.x = graph[boost::source(*it_edge, graph)].pos[0];
85  point_start.y = graph[boost::source(*it_edge, graph)].pos[1];
86  point_start.z = 0;
87  marker.points.push_back(point_start);
88 #endif
89  geometry_msgs::Point point_end;
90  point_end.x = graph[boost::target(*it_edge, graph)].pos[0];
91  point_end.y = graph[boost::target(*it_edge, graph)].pos[1];
92  point_end.z = 0;
93  marker.points.push_back(point_end);
94 
95  // add color
96  std_msgs::ColorRGBA color;
97  color.a = 1.0;
98  color.r = 0;
99  color.g = 0;
100  color.b = 1;
101  marker.colors.push_back(color);
102  marker.colors.push_back(color);
103 #ifdef TRIANGLE
104  marker.colors.push_back(color);
105 #endif
106  }
107 
108 #ifdef TRIANGLE
109  marker.scale.x = 1;
110  marker.scale.y = 1;
111  marker.scale.z = 1;
112 #else
113  marker.scale.x = 0.01;
114 #endif
115  marker.color.a = 1.0;
116  marker.color.r = 0.0;
117  marker.color.g = 1.0;
118  marker.color.b = 0.0;
119 
120  // Now publish edge markers
121  teb_marker_pub_.publish(marker);
122 
123  // Visualize vertices
124  marker.header.frame_id = cfg_->map_frame;
125  marker.header.stamp = ros::Time::now();
126  marker.ns = ns_prefix + "Vertices";
127  marker.id = 0;
128  marker.type = visualization_msgs::Marker::POINTS;
129  marker.action = visualization_msgs::Marker::ADD;
130 
131  GraphVertexIterator it_vert;
132  GraphVertexIterator end_vert;
133  for (boost::tie(it_vert, end_vert) = boost::vertices(graph); it_vert != end_vert; ++it_vert) {
134  geometry_msgs::Point point;
135  point.x = graph[*it_vert].pos[0];
136  point.y = graph[*it_vert].pos[1];
137  point.z = 0;
138  marker.points.push_back(point);
139  // add color
140 
141  std_msgs::ColorRGBA color;
142  color.a = 1.0;
143  if (it_vert == end_vert - 1) {
144  color.r = 1;
145  color.g = 0;
146  color.b = 0;
147  } else {
148  color.r = 0;
149  color.g = 1;
150  color.b = 0;
151  }
152  marker.colors.push_back(color);
153  }
154  // set first color (start vertix) to blue
155  if (!marker.colors.empty()) {
156  marker.colors.front().b = 1;
157  marker.colors.front().g = 0;
158  }
159 
160  marker.scale.x = 0.1;
161  marker.scale.y = 0.1;
162  marker.color.a = 1.0;
163  marker.color.r = 0.0;
164  marker.color.g = 1.0;
165  marker.color.b = 0.0;
166 
167  // Now publish vertex markers
168  teb_marker_pub_.publish(marker);
169 }

171 template <typename BidirIter>
172 void TebVisualization::publishPathContainer(BidirIter first, BidirIter last, const std::string& ns) {
173  if (printErrorWhenNotInitialized()) return;
174 
175  visualization_msgs::Marker marker;
176  marker.header.frame_id = cfg_->map_frame;
177  marker.header.stamp = ros::Time::now();
178  marker.ns = ns;
179  marker.id = 0;
180  marker.type = visualization_msgs::Marker::LINE_LIST;
181  marker.action = visualization_msgs::Marker::ADD;
182 
183  using PathType = typename std::iterator_traits<BidirIter>::value_type; // Get type of the path (point container)
184 
185  // Iterate through path container
186  while (first != last) {
187  // iterate single path points
188  typename PathType::const_iterator it_point;
189  typename PathType::const_iterator end_point;
190  for (it_point = first->begin(), end_point = boost::prior(first->end()); it_point != end_point; ++it_point) {
191  geometry_msgs::Point point_start;
192  point_start.x = get_const_reference(*it_point).x();
193  point_start.y = get_const_reference(*it_point).y();
194  point_start.z = 0;
195  marker.points.push_back(point_start);
196 
197  geometry_msgs::Point point_end;
198  point_end.x = get_const_reference(*boost::next(it_point)).x();
199  point_end.y = get_const_reference(*boost::next(it_point)).y();
200  point_end.z = 0;
201  marker.points.push_back(point_end);
202  }
203  ++first;
204  }
205  marker.scale.x = 0.01;
206  marker.color.a = 1.0;
207  marker.color.r = 0.0;
208  marker.color.g = 1.0;
209  marker.color.b = 0.0;
210 
211  teb_marker_pub_.publish(marker);
212 }
