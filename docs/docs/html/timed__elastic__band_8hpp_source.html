44 bool TimedElasticBand::initTrajectoryToGoal(BidirIter path_start, BidirIter path_end, Fun fun_position, double max_vel_x, double max_vel_theta, boost::optional<double> max_acc_x,
45  boost::optional<double> max_acc_theta, boost::optional<double> start_orientation, boost::optional<double> goal_orientation, int min_samples,
46  bool guess_backwards_motion) {
47  Eigen::Vector2d start_position = fun_position(*path_start);
48  Eigen::Vector2d goal_position = fun_position(*boost::prior(path_end));
49 
50  bool backwards = false;
51 
52  double start_orient;
53  double goal_orient;
54  if (start_orientation) {
55  start_orient = *start_orientation;
56 
57  // check if the goal is behind the start pose (w.r.t. start orientation)
58  if (guess_backwards_motion && (goal_position - start_position).dot(Eigen::Vector2d(std::cos(start_orient), std::sin(start_orient))) < 0) {
59  backwards = true;
60  }
61  } else {
62  Eigen::Vector2d start2goal = goal_position - start_position;
63  start_orient = atan2(start2goal[1], start2goal[0]);
64  }
65 
66  double timestep = 1; // TODO: time
67 
68  if (goal_orientation) {
69  goal_orient = *goal_orientation;
70  } else {
71  goal_orient = start_orient;
72  }
73 
74  if (!isInit()) {
75  addPose(start_position, start_orient, true); // add starting point and mark it as fixed for optimization
76 
77  // we insert middle points now (increase start by 1 and decrease goal by 1)
78  std::advance(path_start, 1);
79  std::advance(path_end, -1);
80  int idx = 0;
81  for (; path_start != path_end; ++path_start) // insert middle-points
82  {
83  // Eigen::Vector2d point_to_goal = path.back()-*it;
84  // double dir_to_goal = atan2(point_to_goal[1],point_to_goal[0]); // direction to goal
85  // Alternative: Direction from last path
86  Eigen::Vector2d curr_point = fun_position(*path_start);
87  Eigen::Vector2d diff_last = curr_point - Pose(idx).position(); // we do not use boost::prior(*path_start) for those cases,
88  // where fun_position() does not return a reference or is expensive.
89  double diff_norm = diff_last.norm();
90 
91  double timestep_vel = diff_norm / max_vel_x; // constant velocity
92  double timestep_acc;
93 
94  if (max_acc_x) {
95  timestep_acc = sqrt(2 * diff_norm / (*max_acc_x)); // constant acceleration
96  if (timestep_vel < timestep_acc && max_acc_x) {
97  timestep = timestep_acc;
98  } else {
99  timestep = timestep_vel;
100  }
101  } else {
102  timestep = timestep_vel;
103  }
104 
105  if (timestep <= 0) {
106  timestep = 0.2; // TODO: this is an assumption
107  }
108 
109  double yaw = atan2(diff_last[1], diff_last[0]);
110  if (backwards) {
111  yaw = g2o::normalize_theta(yaw + M_PI);
112  }
113  addPoseAndTimeDiff(curr_point, yaw, timestep);
114 
115  ++idx;
116  }
117  Eigen::Vector2d diff = goal_position - Pose(idx).position();
118  double diff_norm = diff.norm();
119  double timestep_vel = diff_norm / max_vel_x; // constant velocity
120  if (max_acc_x) {
121  double timestep_acc = sqrt(2 * diff_norm / (*max_acc_x)); // constant acceleration
122  if (timestep_vel < timestep_acc) {
123  timestep = timestep_acc;
124  } else {
125  timestep = timestep_vel;
126  }
127  } else {
128  timestep = timestep_vel;
129  }
130 
131  PoseSE2 goal(goal_position, goal_orient);
132 
133  // if number of samples is not larger than min_samples, insert manually
134  if (sizePoses() < min_samples - 1) {
135  ROS_DEBUG("initTEBtoGoal(): number of generated samples is less than specified by min_samples. Forcing the insertion of more samples...");
136  while (sizePoses() < min_samples - 1) // subtract goal point that will be added later
137  {
138  // Each inserted point bisects the remaining distance. Thus the timestep is also bisected.
139  timestep /= 2;
140  // simple strategy: interpolate between the current pose and the goal
141  addPoseAndTimeDiff(PoseSE2::average(BackPose(), goal), timestep); // let the optimier correct the timestep (TODO: better initialization
142  }
143  }
144 
145  // now add goal
146  addPoseAndTimeDiff(goal, timestep); // add goal point
147  setPoseVertexFixed(sizePoses() - 1, true); // GoalConf is a fixed constraint during optimization
148  } else // size!=0
149  {
150  ROS_WARN("Cannot init TEB between given configuration and goal, because TEB vectors are not empty or TEB is already initialized (call this function before adding states yourself)!");
151  ROS_WARN("Number of TEB configurations: %d, Number of TEB timediffs: %d", sizePoses(), sizeTimeDiffs());
152  return false;
153  }
154  return true;
155 }
