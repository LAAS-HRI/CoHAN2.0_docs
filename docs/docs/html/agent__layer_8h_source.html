157  static double Gaussian2D(double x, double y, double x0, double y0, double A, double varx, double vary) {
158  double dx = x - x0;
159  double dy = y - y0;
160  double d = sqrt((dx * dx) + (dy * dy));
161  double theta = atan2(dy, dx);
162  double xx = d * cos(theta);
163  double yy = d * sin(theta);
164  return A / std::max(d, 1.0) * Guassian1D(xx, 0.0, 1.0, varx) * Guassian1D(yy, 0.0, 1.0, vary);
165  }

179  static double Gaussian2D_skewed(double x, double y, double x0, double y0, double A, double varx, double vary, double skew_ang) {
180  double dx = x - x0;

184  double xx = d * cos(theta - skew_ang);
185  double yy = d * sin(theta - skew_ang);
186  return A / std::max(d, 1.0) * Guassian1D(xx, 0.0, 1.0, varx) * Guassian1D(yy, 0.0, 1.0, vary);
