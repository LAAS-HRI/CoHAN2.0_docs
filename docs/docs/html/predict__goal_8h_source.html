153  void getProbabilities(int id) {
154  std::vector<double> probs; // Temporary storage for probabilities
155  std::vector<double> dists; // Storage for distances to goals
156 
157  if (goal_priors_[id].empty()) {
158  double prior = 1.0 / static_cast<double>(goals_.size());
159  for (int i = 0; i < goals_.size(); i++) {
160  goal_priors_[id].push_back(prior);
161  }
162  }
163 
164  auto trajectory = agents_trajs_[id];
165  int goal_id = 0;
166  double probs_sum = 0;
167  bool still = false;
168 
169  for (auto &goal : goals_) {
170  double probability = 1.0 / goals_.size();
171  int n = trajectory.size();
172  for (int i = 1; i < n; i++) {
173  Eigen::Vector2d heading = trajectory[i] - trajectory[i - 1];
174  if (heading.norm() == 0) {
175  still = true;
176  break;
177  }
178 
179  Eigen::Vector2d goal_vec = goal - trajectory[i];
180  if (i == n - 1) {
181  dists.push_back(goal_vec.norm());
182  }
183 
184  double phi = 0;
185  if (heading.norm() != 0) {
186  phi = std::acos(heading.dot(goal_vec) / (heading.norm() * goal_vec.norm()));
187  }
188 
189  double g = std::exp((i - n) / 0.5);
190  probability *= std::pow(nd_.pdf(phi), g); // Combined proability for the entire trajectory
191  }
192 
193  if (!still) {
194  double goal_probability = goal_priors_[id][goal_id];
195  goal_probability *= probability;
196  probs_sum += goal_probability;
197  probs.push_back(goal_probability);
198  }
199 
200  goal_id++;
201  }
202 
203  if (!still) {
204  for (auto &prob : probs) {
205  prob = prob / probs_sum;
206  double prior = prob;
207  }
208  agent_probs_[id] = probs;
209  }
210  }
