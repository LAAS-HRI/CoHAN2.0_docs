188  _error[0] = fabs(((cos(conf1->theta()) + cos(conf2->theta())) * delta_s[1]) - ((sin(conf1->theta()) + sin(conf2->theta())) * delta_s[0]));
189 
190  // limit minimum turning radius
191  double angle_diff = g2o::normalize_theta(conf2->theta() - conf1->theta());
192  if (angle_diff == 0)
193  _error[1] = 0; // straight line motion
194  else if (cfg_->trajectory.exact_arc_length) // use exact computation of the radius
195  _error[1] = penaltyBoundFromBelow(fabs(delta_s.norm() / (2 * sin(angle_diff / 2))), cfg_->robot.min_turning_radius, 0.0);
196  else
197  _error[1] = penaltyBoundFromBelow(delta_s.norm() / fabs(angle_diff), cfg_->robot.min_turning_radius, 0.0);
