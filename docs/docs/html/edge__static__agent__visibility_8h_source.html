69  void computeError() override {
70  ROS_ASSERT_MSG(cfg_, "You must call setParameters() on EdgeStaticAgentVisibility()");
71  const auto *robot_bandpt = static_cast<const VertexPose *>(_vertices[0]);
72 
73  Eigen::Vector2d d_rtoh = static_agent_.position() - robot_bandpt->position();
74  Eigen::Vector2d d_htor = robot_bandpt->position() - static_agent_.position();
75  Eigen::Vector2d agent_look_at = {cos(static_agent_.theta()), sin(static_agent_.theta())};
76  Eigen::Vector2d robot_look_at = {cos(robot_bandpt->theta()), sin(robot_bandpt->theta())};
77 
78  // calculate the angle between human heading and direction vector from human to robot
79  double delta_psi = fabs(acos(agent_look_at.dot(d_htor) / (agent_look_at.norm() * d_htor.norm())));
80  double c_visibility = 0.0;
81  double ang = agent_look_at.dot(robot_look_at);
82 
83  // Apply the cost if the robot is not within the FOV of human (behind the human)
84  if (delta_psi >= cfg_->agent.fov * M_PI / 180) {
85  if (ang >= 0) {
86  c_visibility = 5 * ((std::pow(2, -(std::pow(d_rtoh.x(), 2)))) * (std::pow(2, -(std::pow(d_rtoh.y(), 2)))));
87  } else {
88  c_visibility = 0.;
89  }
90  }
91 
92  _error[0] = penaltyBoundFromAbove(c_visibility, cfg_->hateb.visibility_cost_threshold, cfg_->optim.penalty_epsilon);
93 
94  ROS_ASSERT_MSG(std::isfinite(_error[0]), "EdgeStaticAgentVisibility::computeError() _error[0]=%f\n", _error[0]);
95  }
