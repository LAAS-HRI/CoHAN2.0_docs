261  void rotateGlobal(double angle, bool adjust_theta = true) {
262  double new_x = (std::cos(angle) * position_.x()) - (std::sin(angle) * position_.y());
263  double new_y = (std::sin(angle) * position_.x()) + (std::cos(angle) * position_.y());
