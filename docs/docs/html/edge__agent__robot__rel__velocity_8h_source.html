68  void computeError() override {
69  ROS_ASSERT_MSG(cfg_, "You must call setHATebConfig() on EdgeAgentRobotRelVelocity()");
70  const auto *robot_bandpt = static_cast<const VertexPose *>(_vertices[0]);
71  const auto *robot_bandpt_nxt = static_cast<const VertexPose *>(_vertices[1]);
72  const auto *dt_robot = static_cast<const VertexTimeDiff *>(_vertices[2]);
73  const auto *agent_bandpt = static_cast<const VertexPose *>(_vertices[3]);
74  const auto *agent_bandpt_nxt = static_cast<const VertexPose *>(_vertices[4]);
75  const auto *dt_agent = static_cast<const VertexTimeDiff *>(_vertices[5]);
76 
77  // Get the velocities of the robot and agent
78  Eigen::Vector2d diff_robot = robot_bandpt_nxt->position() - robot_bandpt->position();
79  Eigen::Vector2d robot_vel = diff_robot / dt_robot->dt();
80  Eigen::Vector2d diff_agent = agent_bandpt_nxt->position() - agent_bandpt->position();
81  Eigen::Vector2d agent_vel = diff_agent / dt_agent->dt();
82 
83  // Get the direction vectors
84  Eigen::Vector2d d_rtoh = agent_bandpt->position() - robot_bandpt->position();
85  Eigen::Vector2d d_htor = robot_bandpt->position() - agent_bandpt->position();
86 
87  // Cost definition
88  double rel_vel_cost = (std::max((robot_vel - agent_vel).dot(d_rtoh), 0.0) + robot_vel.norm() + 1) / d_rtoh.norm();
89 
90  _error[0] = penaltyBoundFromAbove(rel_vel_cost, cfg_->hateb.rel_vel_cost_threshold, cfg_->optim.penalty_epsilon);
91 
92  ROS_ASSERT_MSG(std::isfinite(_error[0]), "EdgeAgentRobot::computeError() _error[0]=%f\n", _error[0]);
93  }
