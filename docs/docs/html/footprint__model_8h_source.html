186  void visualizeModel(const PoseSE2& current_pose, std::vector<visualization_msgs::Marker>& markers, const std_msgs::ColorRGBA& color) const override {
187  // point footprint
188  markers.emplace_back();
189  visualization_msgs::Marker& marker = markers.back();
190  marker.type = visualization_msgs::Marker::POINTS;
191  current_pose.toPoseMsg(marker.pose); // all points are transformed into the robot/human frame!
192  marker.points.emplace_back();
193  marker.scale.x = 0.025;
194  marker.color = color;
195 
196  if (min_obstacle_dist_ <= 0) {
197  return;
198  }
199 
200  // footprint with min_obstacle_dist
201  markers.emplace_back();
202  visualization_msgs::Marker& marker2 = markers.back();
203  marker2.type = visualization_msgs::Marker::LINE_STRIP;
204  marker2.scale.x = 0.025;
205  marker2.color = color;
206  current_pose.toPoseMsg(marker2.pose); // all points are transformed into the robot/human frame!
207 
208  const double n = 9;
209  const double r = min_obstacle_dist_;
210  for (double theta = 0; theta <= 2 * M_PI; theta += M_PI / n) {
211  geometry_msgs::Point pt;
212  pt.x = r * cos(theta);
213  pt.y = r * sin(theta);
214  marker2.points.push_back(pt);
215  }
216  }

218  private:
219  const double min_obstacle_dist_ = 0.0;

338  double calculateDistance(const PoseSE2& current_pose, const Obstacle* obstacle) const override {
339  Eigen::Vector2d dir = current_pose.orientationUnitVec();
340  double dist_front = obstacle->getMinimumDistance(current_pose.position() + front_offset_ * dir) - front_radius_;
341  double dist_rear = obstacle->getMinimumDistance(current_pose.position() - rear_offset_ * dir) - rear_radius_;
342  return std::min(dist_front, dist_rear);
343  }

352  double estimateSpatioTemporalDistance(const PoseSE2& current_pose, const Obstacle* obstacle, double t) const override {

355  double dist_rear = obstacle->getMinimumSpatioTemporalDistance(current_pose.position() - rear_offset_ * dir, t) - rear_radius_;
356  return std::min(dist_front, dist_rear);
357  }

376  marker1.pose.position.y += front_offset_ * dir.y();
377  marker1.scale.x = marker1.scale.y = 2 * front_radius_; // scale = diameter

379  marker1.color = color;

385  current_pose.toPoseMsg(marker2.pose);

387  marker2.pose.position.y -= rear_offset_ * dir.y();

391  }

563 class PolygonFootprint : public BaseFootprintModel {
564  public:
569  explicit PolygonFootprint(Point2dContainer vertices) : vertices_(std::move(vertices)) {}
570 
574  ~PolygonFootprint() override = default;
575 
580  void setVertices(const Point2dContainer& vertices) { vertices_ = vertices; }
581 
588  double calculateDistance(const PoseSE2& current_pose, const Obstacle* obstacle) const override {
589  Point2dContainer polygon_world(vertices_.size());
590  transformToWorld(current_pose, polygon_world);
591  return obstacle->getMinimumDistance(polygon_world);
592  }
593 
601  double estimateSpatioTemporalDistance(const PoseSE2& current_pose, const Obstacle* obstacle, double t) const override {
602  Point2dContainer polygon_world(vertices_.size());
603  transformToWorld(current_pose, polygon_world);
604  return obstacle->getMinimumSpatioTemporalDistance(polygon_world, t);
605  }
606 
616  void visualizeModel(const PoseSE2& current_pose, std::vector<visualization_msgs::Marker>& markers, const std_msgs::ColorRGBA& color) const override {
617  if (vertices_.empty()) return;
618 
619  markers.emplace_back();
620  visualization_msgs::Marker& marker = markers.front();
621  marker.type = visualization_msgs::Marker::LINE_STRIP;
622  current_pose.toPoseMsg(marker.pose); // all points are transformed into the robot/human frame!
623 
624  for (const auto& vertice : vertices_) {
625  geometry_msgs::Point point;
626  point.x = vertice.x();
627  point.y = vertice.y();
628  point.z = 0;
629  marker.points.push_back(point);
630  }
631  // add first point again in order to close the polygon
632  geometry_msgs::Point point;
633  point.x = vertices_.front().x();
634  point.y = vertices_.front().y();
635  point.z = 0;
636  marker.points.push_back(point);
637 
638  marker.scale.x = 0.025;
639  marker.color = color;
640  }
641 
646  double getInscribedRadius() override {
647  double min_dist = std::numeric_limits<double>::max();
648  Eigen::Vector2d center(0.0, 0.0);
649 
650  if (vertices_.size() <= 2) {
651  return 0.0;
652  }
653 
654  for (int i = 0; i < static_cast<int>(vertices_.size()) - 1; ++i) {
655  // compute distance from the robot/human center point to the first vertex
656  double vertex_dist = vertices_[i].norm();
657  double edge_dist = distance_point_to_segment_2d(center, vertices_[i], vertices_[i + 1]);
658  min_dist = std::min({min_dist, vertex_dist, edge_dist});
659  }
660 
661  // we also need to check the last vertex and the first vertex
662  double vertex_dist = vertices_.back().norm();
663  double edge_dist = distance_point_to_segment_2d(center, vertices_.back(), vertices_.front());
664  return std::min({min_dist, vertex_dist, edge_dist});
665  }
666 
671  double getCircumscribedRadius() const override {
672  double radius = 0.0;
673  for (const auto& vertex : vertices_) {
674  double dist = std::hypot(vertex.x(), vertex.y());
675  radius = std::max(radius, dist);
676  }
677  return radius;
678  }
679 
680  private:
686  void transformToWorld(const PoseSE2& current_pose, Point2dContainer& polygon_world) const {
687  double cos_th = std::cos(current_pose.theta());
688  double sin_th = std::sin(current_pose.theta());
689  for (std::size_t i = 0; i < vertices_.size(); ++i) {
690  polygon_world[i].x() = current_pose.x() + cos_th * vertices_[i].x() - sin_th * vertices_[i].y();
691  polygon_world[i].y() = current_pose.y() + sin_th * vertices_[i].x() + cos_th * vertices_[i].y();
692  }
693  }
694 
695  Point2dContainer vertices_;
696 };
