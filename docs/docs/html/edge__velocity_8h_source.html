101  const double angle_diff = g2o::normalize_theta(conf2->theta() - conf1->theta());
102  if (cfg_->trajectory.exact_arc_length && angle_diff != 0) {
103  double radius = dist / (2 * sin(angle_diff / 2));
104  dist = fabs(angle_diff * radius); // actual arg length!

110  const double omega = angle_diff / delta_t->estimate();

137  void linearizeOplus() {
138  ROS_ASSERT_MSG(cfg_, "You must call setHATebConfig on EdgeVelocity()");
139  const VertexPose *conf1 = static_cast<const VertexPose *>(_vertices[0]);
140  const VertexPose *conf2 = static_cast<const VertexPose *>(_vertices[1]);
141  const VertexTimeDiff *deltaT = static_cast<const VertexTimeDiff *>(_vertices[2]);
142 
143  Eigen::Vector2d deltaS = conf2->position() - conf1->position();
144  double dist = deltaS.norm();
145  double aux1 = dist * deltaT->estimate();
146  double aux2 = 1 / deltaT->estimate();
147 
148  double vel = dist * aux2;
149  double omega = g2o::normalize_theta(conf2->theta() - conf1->theta()) * aux2;
150 
151  double dev_border_vel = penaltyBoundToIntervalDerivative(vel, -cfg_->robot.max_vel_x_backwards, cfg_->robot.max_vel_x, cfg_->optim.penalty_epsilon);
152  double dev_border_omega = penaltyBoundToIntervalDerivative(omega, cfg_->robot.max_vel_theta, cfg_->optim.penalty_epsilon);
153 
154  _jacobianOplus[0].resize(2, 3); // conf1
155  _jacobianOplus[1].resize(2, 3); // conf2
156  _jacobianOplus[2].resize(2, 1); // deltaT
157 
158  // if (aux1==0) aux1=1e-6;
159  // if (aux2==0) aux2=1e-6;
160 
161  if (dev_border_vel != 0) {
162  double aux3 = dev_border_vel / aux1;
163  _jacobianOplus[0](0, 0) = -deltaS[0] * aux3; // vel x1
164  _jacobianOplus[0](0, 1) = -deltaS[1] * aux3; // vel y1
165  _jacobianOplus[1](0, 0) = deltaS[0] * aux3; // vel x2
166  _jacobianOplus[1](0, 1) = deltaS[1] * aux3; // vel y2
167  _jacobianOplus[2](0, 0) = -vel * aux2 * dev_border_vel; // vel deltaT
168  } else {
169  _jacobianOplus[0](0, 0) = 0; // vel x1
170  _jacobianOplus[0](0, 1) = 0; // vel y1
171  _jacobianOplus[1](0, 0) = 0; // vel x2
172  _jacobianOplus[1](0, 1) = 0; // vel y2
173  _jacobianOplus[2](0, 0) = 0; // vel deltaT
174  }
175 
176  if (dev_border_omega != 0) {
177  double aux4 = aux2 * dev_border_omega;
178  _jacobianOplus[2](1, 0) = -omega * aux4; // omega deltaT
179  _jacobianOplus[0](1, 2) = -aux4; // omega angle1
180  _jacobianOplus[1](1, 2) = aux4; // omega angle2
181  } else {
182  _jacobianOplus[2](1, 0) = 0; // omega deltaT
183  _jacobianOplus[0](1, 2) = 0; // omega angle1
184  _jacobianOplus[1](1, 2) = 0; // omega angle2
185  }

188  _jacobianOplus[0](1, 1) = 0; // omega y1
189  _jacobianOplus[1](1, 0) = 0; // omega x2

191  _jacobianOplus[0](0, 2) = 0; // vel angle1
