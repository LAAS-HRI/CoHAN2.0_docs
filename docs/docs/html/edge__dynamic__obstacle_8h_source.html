87  void computeError() override {
88  ROS_ASSERT_MSG(cfg_ && _measurement, "You must call setHATebConfig() on EdgeDynamicObstacle()");
89  const auto* bandpt = static_cast<const VertexPose*>(_vertices[0]);
90 
91  double dist = 0;
92 
93  if (type_ == static_cast<int>(cohan_msgs::AgentType::ROBOT)) {
94  dist = cfg_->robot_model->estimateSpatioTemporalDistance(bandpt->pose(), _measurement, t_);
95  }
96 
97  if (type_ == static_cast<int>(cohan_msgs::AgentType::HUMAN)) {
98  dist = cfg_->human_model->estimateSpatioTemporalDistance(bandpt->pose(), _measurement, t_);
99  }
100 
101  _error[0] = penaltyBoundFromBelow(dist, cfg_->obstacles.min_obstacle_dist, cfg_->optim.penalty_epsilon);
102  _error[1] = penaltyBoundFromBelow(dist, cfg_->obstacles.dynamic_obstacle_inflation_dist, 0.0);
103 
104  ROS_ASSERT_MSG(std::isfinite(_error[0]), "EdgeDynamicObstacle::computeError() _error[0]=%f\n", _error[0]);
105  }
