133  bool worldToMap(double wx, double wy, int& mx, int& my) const {
134  if (wx < origin_x_ || wy < origin_y_) return false;
135 
136  mx = static_cast<int>((wx - origin_x_) / resolution_);
137  my = static_cast<int>((wy - origin_y_) / resolution_);
138 
139  return mx < size_x_ && my < size_y_;
140  }

158  static Point getLeftPoint(Point p1, Point p2, Point p3, double dist = 1) {
159  double x = p2.first - p1.first;
160  double y = p2.second - p1.second;
161  double point_dist = std::hypot(x, y);
162 
163  Point p;
164  p.first = p3.first - (dist * y / point_dist);
165  p.second = p3.second + (dist * x / point_dist);
166  return p;
167  }
168 
177  static Point getRightPoint(Point p1, Point p2, Point p3, double dist = 1) {
178  double x = p2.first - p1.first;
179  double y = p2.second - p1.second;
180  double point_dist = std::hypot(x, y);
181 
182  Point p;
183  p.first = p3.first + (dist * y / point_dist);
184  p.second = p3.second - (dist * x / point_dist);
185  return p;
186  }

195  static std::vector<Point> getTwoPoints(Point p1, Point p2, double radius) {
196  std::vector<Point> points;
197  auto l_p = getLeftPoint(p1, p2, p2, radius);
198  points.push_back(l_p);
199  auto r_p = getRightPoint(p1, p2, p2, radius);
200  points.push_back(r_p);
201  return points;
202  }
