94  const Eigen::Vector2d diff2 = pose3->position() - pose2->position();
95

97  double dist2 = diff2.norm();

99  const double angle_diff2 = g2o::normalize_theta(pose3->theta() - pose2->theta());

106  }

109  dist2 = fabs(angle_diff2 * radius); // actual arg length!

114  double vel2 = dist2 / dt2->dt();

446  double sin_theta2 = std::sin(pose2->theta());

610  double sin_theta1 = std::sin(pose_pre_goal->theta());
611 
612  // transform pose2 into robot frame pose1 (inverse 2d rotation matrix)
613  double p1_dx = (cos_theta1 * diff.x()) + (sin_theta1 * diff.y());
614  double p1_dy = (-sin_theta1 * diff.x()) + (cos_theta1 * diff.y());
615 
616  double vel1_x = p1_dx / dt->dt();
617  double vel1_y = p1_dy / dt->dt();
618  double vel2_x = _measurement->linear.x;
619  double vel2_y = _measurement->linear.y;
620 
621  double acc_lin_x = (vel2_x - vel1_x) / dt->dt();
622  double acc_lin_y = (vel2_y - vel1_y) / dt->dt();
623 
624  _error[0] = penaltyBoundToInterval(acc_lin_x, cfg_->robot.acc_lim_x, cfg_->optim.penalty_epsilon);
625  _error[1] = penaltyBoundToInterval(acc_lin_y, cfg_->robot.acc_lim_y, cfg_->optim.penalty_epsilon);
626 
627  // ANGULAR ACCELERATION
628  double omega1 = g2o::normalize_theta(pose_goal->theta() - pose_pre_goal->theta()) / dt->dt();
629  double omega2 = _measurement->angular.z;
630  double acc_rot = (omega2 - omega1) / dt->dt();
631 
632  _error[2] = penaltyBoundToInterval(acc_rot, cfg_->robot.acc_lim_theta, cfg_->optim.penalty_epsilon);
633 
634  ROS_ASSERT_MSG(std::isfinite(_error[0]), "EdgeAccelerationGoal::computeError() translational: _error[0]=%f\n", _error[0]);
635  ROS_ASSERT_MSG(std::isfinite(_error[1]), "EdgeAccelerationGoal::computeError() strafing: _error[1]=%f\n", _error[1]);
636  ROS_ASSERT_MSG(std::isfinite(_error[2]), "EdgeAccelerationGoal::computeError() rotational: _error[2]=%f\n", _error[2]);
637  }
638 
643  void setGoalVelocity(const geometry_msgs::Twist& vel_goal) { _measurement = &vel_goal; }
