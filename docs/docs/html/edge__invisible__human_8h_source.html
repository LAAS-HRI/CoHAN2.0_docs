78  void computeError() override {
79  ROS_ASSERT_MSG(cfg_ && _measurement, "You must call setHATebConfig() on EdgeInvisibleHuman()");
80  const auto* bandpt = static_cast<const VertexPose*>(_vertices[0]);
81  const auto* bandpt_nxt = static_cast<const VertexPose*>(_vertices[1]);
82  const auto* dt = static_cast<const VertexTimeDiff*>(_vertices[2]);
83 
84  // Get the distance and the velocity
85  double dist = cfg_->robot_model->calculateDistance(bandpt->pose(), _measurement);
86  auto robot_vel = (bandpt_nxt->position() - bandpt->position()) / dt->dt();
87 
88  double cost = dist / (V_i_ + robot_vel.norm() - 0.1);
89  if (t_ > 0.5) { // Accounting for human reaction time
90  // Cost calculation
91  cost = dist / (std::max(V_i_ - (a_norm_ * t_) + robot_vel.norm() - 0.1, 0.01));
92  }
93  _error[0] = penaltyBoundFromBelow(cost, cfg_->hateb.invisible_human_threshold, cfg_->optim.penalty_epsilon);
94 
95  ROS_ASSERT_MSG(std::isfinite(_error[0]), "EdgeInvisibleHuman::computeError() _error[0]=%f\n", _error[0]);
96  }
