560  void lookupTwist(const std::string &tracking_frame, const std::string &observation_frame, const std::string &reference_frame, const tf2::Vector3 &reference_point,
561  const std::string &reference_point_frame, const ros::Time &time, const ros::Duration &averaging_interval, geometry_msgs::Twist &twist) const {
562  ros::Time latest_time;
563  ros::Time target_time;
564 
565  tf2::CompactFrameID target_id = tf_->_lookupFrameNumber(tf::strip_leading_slash(tracking_frame));
566  tf2::CompactFrameID source_id = tf_->_lookupFrameNumber(tf::strip_leading_slash(observation_frame));
567  tf_->_getLatestCommonTime(source_id, target_id, latest_time, nullptr);
568 
569  if (ros::Time() == time) {
570  target_time = latest_time;
571  } else {
572  target_time = time;
573  }
574 
575  ros::Time end_time = std::min(target_time + averaging_interval * 0.5, latest_time);
576 
577  ros::Time start_time = std::max(ros::Time().fromSec(.00001) + averaging_interval, end_time) - averaging_interval; // don't collide with zero
578  ros::Duration corrected_averaging_interval = end_time - start_time; // correct for the possiblity that start time was
579  // truncated above.
580  geometry_msgs::TransformStamped start_msg;
581  geometry_msgs::TransformStamped end_msg;
582  start_msg = tf_->lookupTransform(observation_frame, tracking_frame, start_time);
583  end_msg = tf_->lookupTransform(observation_frame, tracking_frame, end_time);
584 
585  tf2::Stamped<tf2::Transform> start;
586  tf2::Stamped<tf2::Transform> end;
587  tf2::fromMsg(start_msg, start);
588  tf2::fromMsg(end_msg, end);
589 
590  tf2::Matrix3x3 temp = start.getBasis().inverse() * end.getBasis();
591  tf2::Quaternion quat_temp;
592  temp.getRotation(quat_temp);
593  tf2::Vector3 o = start.getBasis() * quat_temp.getAxis();
594  tfScalar ang = quat_temp.getAngle();
595 
596  double delta_x = end.getOrigin().getX() - start.getOrigin().getX();
597  double delta_y = end.getOrigin().getY() - start.getOrigin().getY();
598  double delta_z = end.getOrigin().getZ() - start.getOrigin().getZ();
599 
600  tf2::Vector3 twist_vel((delta_x) / corrected_averaging_interval.toSec(), (delta_y) / corrected_averaging_interval.toSec(), (delta_z) / corrected_averaging_interval.toSec());
601  tf2::Vector3 twist_rot = o * (ang / corrected_averaging_interval.toSec());
602 
603  // This is a twist w/ reference frame in observation_frame and reference
604  // point is in the tracking_frame at the origin (at start_time)
605 
606  // correct for the position of the reference frame
607  tf2::Stamped<tf2::Transform> inverse;
608  tf2::fromMsg(tf_->lookupTransform(reference_frame, tracking_frame, target_time), inverse);
609  tf2::Vector3 out_rot = inverse.getBasis() * twist_rot;
610  tf2::Vector3 out_vel = inverse.getBasis() * twist_vel + inverse.getOrigin().cross(out_rot);
611 
612  // Rereference the twist about a new reference point
613  // Start by computing the original reference point in the reference frame:
614  tf2::Stamped<tf2::Vector3> rp_orig(tf2::Vector3(0, 0, 0), target_time, tracking_frame);
615  geometry_msgs::TransformStamped reference_frame_trans;
616  tf2::fromMsg(tf_->lookupTransform(reference_frame, rp_orig.frame_id_, rp_orig.stamp_), reference_frame_trans);
617 
618  geometry_msgs::PointStamped rp_orig_msg;
619  tf2::toMsg(rp_orig, rp_orig_msg);
620  tf2::doTransform(rp_orig_msg, rp_orig_msg, reference_frame_trans);
621 
622  // convert the requrested reference point into the right frame
623  tf2::Stamped<tf2::Vector3> rp_desired(reference_point, target_time, reference_point_frame);
624  geometry_msgs::PointStamped rp_desired_msg;
625  tf2::toMsg(rp_desired, rp_desired_msg);
626  tf2::doTransform(rp_desired_msg, rp_desired_msg, reference_frame_trans);
627  // compute the delta
628  tf2::Vector3 delta = rp_desired - rp_orig;
629  // Correct for the change in reference point
630  out_vel = out_vel + out_rot * delta;
631  // out_rot unchanged
632 
633  /*
634  printf("KDL: Rotation %f %f %f, Translation:%f %f %f\n",
635  out_rot.x(),out_rot.y(),out_rot.z(),
636  out_vel.x(),out_vel.y(),out_vel.z());
637  */
638 
639  twist.linear.x = out_vel.x();
640  twist.linear.y = out_vel.y();
641  twist.linear.z = out_vel.z();
642  twist.angular.x = out_rot.x();
643  twist.angular.y = out_rot.y();
644  twist.angular.z = out_rot.z();
645  }
